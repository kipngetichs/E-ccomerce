-- Delivery Status Classification (Calculated Column)
Delivery Status =
SWITCH(
    TRUE(),
    olist_orders[order_delivered_customer_date] = BLANK(),
        "Not Yet Delivered",
    olist_orders[order_delivered_customer_date]
        <= olist_orders[order_estimated_delivery_date],

        olist_orders[order_estimated_delivery_date],
        olist_orders[order_delivered_customer_date],

DIVIDE(
    CALCULATE(
        COUNTROWS(olist_orders),
        olist_orders[Delivery Status] = "On-Time / Early",
        olist_orders[order_status] = "delivered"
    ),
    CALCULATE(
        COUNTROWS(olist_orders),
        olist_orders[order_status] = "delivered"
    ),
    0

                olist_orders[order_id], BOTH)

## SQL Queries — The Foundation Beneath the Dashboard

<details>
<summary><strong>Query 1 — Delivery Performance & Review Score Impact</strong> — click to expand</summary>

```sql
SELECT
    CASE
        WHEN o.order_delivered_customer_date IS NULL
            THEN 'Not Yet Delivered'
        WHEN o.order_delivered_customer_date
            <= o.order_estimated_delivery_date
            THEN 'On-Time / Early'
        WHEN DATEDIFF(DAY,
            o.order_estimated_delivery_date,
            o.order_delivered_customer_date) <= 3
            THEN 'Slightly Late (1-3 days)'
        ELSE 'Significantly Late (4+ days)'
    END                                              AS delivery_status,
    COUNT(DISTINCT o.order_id)                       AS total_orders,
    ROUND(AVG(CAST(r.review_score AS FLOAT)), 2)     AS avg_review_score,
    ROUND(
        100.0 * SUM(CASE WHEN r.review_score = 1
                    THEN 1 ELSE 0 END)
              / NULLIF(COUNT(r.review_score), 0), 1
    )                                                AS pct_one_star_reviews,
    ROUND(AVG(
        CASE
            WHEN o.order_delivered_customer_date
                 > o.order_estimated_delivery_date
            THEN DATEDIFF(DAY,
                o.order_estimated_delivery_date,
                o.order_delivered_customer_date)
        END
    ), 1)                                            AS avg_days_late
FROM olist_orders o
LEFT JOIN olist_order_reviews r ON o.order_id = r.order_id
WHERE o.order_status = 'delivered'
GROUP BY
    CASE
        WHEN o.order_delivered_customer_date IS NULL
            THEN 'Not Yet Delivered'
        WHEN o.order_delivered_customer_date
            <= o.order_estimated_delivery_date
            THEN 'On-Time / Early'
        WHEN DATEDIFF(DAY,
            o.order_estimated_delivery_date,
            o.order_delivered_customer_date) <= 3
            THEN 'Slightly Late (1-3 days)'
        ELSE 'Significantly Late (4+ days)'
    END
ORDER BY avg_review_score DESC;
```

<summary><strong>Query 2 — Seller Scorecard: Revenue, Satisfaction & Dispatch</strong> — click to expand</summary>

```sql
SELECT
    s.seller_city,
    s.seller_state,
    s.seller_id,
    COUNT(DISTINCT oi.order_id)                      AS total_orders,
    ROUND(SUM(oi.price), 2)                          AS total_revenue,
    ROUND(AVG(CAST(r.review_score AS FLOAT)), 2)     AS avg_review_score,
    ROUND(
        100.0 * SUM(
            CASE WHEN o.order_delivered_carrier_date
                      <= oi.shipping_limit_date
                 THEN 1 ELSE 0 END)
              / NULLIF(COUNT(oi.order_id), 0), 1
    )                                                AS dispatch_compliance_pct,
    CASE
        WHEN AVG(CAST(r.review_score AS FLOAT)) >= 4.5
         AND (100.0 * SUM(CASE WHEN o.order_delivered_carrier_date
              <= oi.shipping_limit_date THEN 1 ELSE 0 END)
              / NULLIF(COUNT(oi.order_id),0)) >= 90
            THEN 'GOLD - Top Performer'
        WHEN AVG(CAST(r.review_score AS FLOAT)) >= 3.5
         AND (100.0 * SUM(CASE WHEN o.order_delivered_carrier_date
              <= oi.shipping_limit_date THEN 1 ELSE 0 END)
              / NULLIF(COUNT(oi.order_id),0)) >= 75
            THEN 'SILVER - Solid Performer'
        WHEN AVG(CAST(r.review_score AS FLOAT)) >= 2.5
            THEN 'BRONZE - Needs Improvement'
        ELSE 'AT RISK - Urgent Review'
    END                                              AS performance_tier
FROM olist_order_items oi
JOIN olist_orders o       ON oi.order_id  = o.order_id
JOIN olist_sellers s      ON oi.seller_id = s.seller_id
LEFT JOIN olist_order_reviews r ON o.order_id = r.order_id
WHERE o.order_status = 'delivered'
GROUP BY s.seller_id, s.seller_city, s.seller_state
HAVING COUNT(DISTINCT oi.order_id) >= 10
ORDER BY total_revenue DESC;
```

<summary><strong>Query 3 — Category Revenue, Freight & Net Satisfaction Index</strong> — click to expand</summary>

```sql
SELECT
    COALESCE(ct.product_category_name_english,
             p.product_category_name)                AS category_english,
    COUNT(DISTINCT oi.order_id)                      AS total_orders,
    ROUND(SUM(oi.price), 2)                          AS category_revenue,
    ROUND(AVG(oi.price), 2)                          AS avg_item_price,
    ROUND(
        100.0 * SUM(oi.freight_value)
              / NULLIF(SUM(oi.price), 0), 1
    )                                                AS freight_pct_of_revenue,
    ROUND(AVG(CAST(r.review_score AS FLOAT)), 2)     AS avg_review_score,
    ROUND(
        100.0 * (
            SUM(CASE WHEN r.review_score >= 4 THEN 1 ELSE 0 END) -
            SUM(CASE WHEN r.review_score <= 2 THEN 1 ELSE 0 END)
        ) / NULLIF(COUNT(r.review_score), 0), 1
    )                                                AS net_satisfaction_index
FROM olist_order_items oi
JOIN olist_products p ON oi.product_id = p.product_id
JOIN olist_orders o   ON oi.order_id   = o.order_id
LEFT JOIN product_category_name_translation ct
    ON p.product_category_name = ct.product_category_name
LEFT JOIN olist_order_reviews r ON o.order_id = r.order_id
WHERE o.order_status = 'delivered'
GROUP BY ct.product_category_name_english, p.product_category_name
ORDER BY category_revenue DESC;
```

<summary><strong>Query 4 — Monthly GMV, AOV & Payment Behaviour</strong> — click to expand</summary>

```sql
SELECT
    FORMAT(o.order_purchase_timestamp, 'yyyy-MM')    AS year_month,
    COUNT(DISTINCT o.order_id)                       AS total_orders,
    COUNT(DISTINCT c.customer_unique_id)             AS unique_customers,
    ROUND(SUM(p.payment_value), 2)                   AS monthly_gmv,
    ROUND(AVG(p.payment_value), 2)                   AS avg_order_value,
    ROUND(
        100.0 * SUM(CASE WHEN p.payment_type = 'credit_card'
                    THEN p.payment_value ELSE 0 END)
              / NULLIF(SUM(p.payment_value), 0), 1
    )                                                AS credit_card_share_pct,
    ROUND(
        100.0 * SUM(CASE WHEN p.payment_type = 'boleto'
                    THEN p.payment_value ELSE 0 END)
              / NULLIF(SUM(p.payment_value), 0), 1
    )                                                AS boleto_share_pct,
    ROUND(
        AVG(CAST(p.payment_installments AS FLOAT)), 1
    )                                                AS avg_installments
FROM olist_orders o
JOIN olist_order_payments p ON o.order_id    = p.order_id
JOIN olist_customers c      ON o.customer_id = c.customer_id
WHERE o.order_status NOT IN ('canceled', 'unavailable')
  AND FORMAT(o.order_purchase_timestamp, 'yyyy-MM') <> '2018-09'
GROUP BY FORMAT(o.order_purchase_timestamp, 'yyyy-MM')
ORDER BY year_month;
```
